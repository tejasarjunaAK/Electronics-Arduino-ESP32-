int in2=3;
int en1=6;

  pinMode(in1,OUTPUT);
  pinMode(in2,OUTPUT);
  pinMode(in3,OUTPUT);
  pinMode(in4,OUTPUT);
  pinMode(en1,OUTPUT);
  pinMode(en3,OUTPUT);}

  digitalWrite(in2,HIGH);
  analogWrite(en1,128);

  digitalWrite(in4,HIGH);

  digitalWrite(in2,LOW);

  analogWrite(en3,0);

  digitalWrite(in1,LOW);
  digitalWrite(in2,HIGH);
  analogWrite(en1,128);
  digitalWrite(in3,HIGH);
  digitalWrite(in4,LOW);
  analogWrite(en3,128);
  delay(1000);
  analogWrite(en1,0);
  analogWrite(en3,0);
  delay(1000);
  digitalWrite(in1,HIGH);
  digitalWrite(in2,LOW);
  analogWrite(en1,128);
  digitalWrite(in3,LOW);
  digitalWrite(in4,HIGH);
